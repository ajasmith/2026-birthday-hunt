% Build encoding map programmatically by walking both lists in parallel
\makeatletter
\def\enc@buildmap#1,#2\enc@sep#3,#4\enc@sep{%
  \expandafter\gdef\csname encode@#1\endcsname{#3}%
  \enc@cont{#2}{#4}%
}
\def\enc@cont#1#2{%
  \ifx\relax#1\else
    \enc@buildmap#1\enc@sep#2\enc@sep
  \fi
}
\edef\enc@call{%
  \noexpand\enc@buildmap
  \decodedletters,\noexpand\relax\noexpand\enc@sep
  \codedletters,\noexpand\relax\noexpand\enc@sep
}
\enc@call
\makeatother

% Look up encoded character for a given decoded letter
\newcommand{\encodechar}[1]{%
  \IfStrEq{#1}{ }{~}{\csname encode@#1\endcsname}%
}

\newcount\msgidx
\newcommand{\foreachmessagechar}[1]{%
  \StrLen{\decodedMessage}[\decodedMessageLen]%
  \msgidx=1
  \loop
    \ifnum\msgidx>\decodedMessageLen
    \else
      \StrChar{\decodedMessage}{\the\msgidx}[\msgchar]%
      #1%
      \advance\msgidx by 1
  \repeat
}
